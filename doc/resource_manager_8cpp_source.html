125  void ResourceManager::populateCharacters(std::unordered_map<std::string, unsigned int> fontsAndSizes)
126  {
127  FT_Library ft;
128  FT_Face face;
129 
130  // Set up memory for all textures
131  unsigned char * texMemory;
132  int memH = 1024;
133  int memW = 1024;
134  int usedX = 0;
135  int usedY = 0;
136  // Any bigger any we'll need to talk to the GPU to see what t's max texture size is
137  // Make sure memory is initialised
138  texMemory = (unsigned char *)malloc(memW * memH);
139  memset(texMemory, 0, memW * memH);
140 
141  std::vector<glm::ivec2> glyphSizes(fontsAndSizes.size());
142  std::vector<Character> chars(m_ASCIIend - m_ASCIIstart + 1);
143 
144  // Init freetype
145  if (FT_Init_FreeType(&ft)) ENGINE_FATAL("Error: Could not start FreeType");
146 
147  int startY = 0;
148  for (auto it = fontsAndSizes.begin(); it != fontsAndSizes.end(); ++it)
149  {
150  int i = 0;
151  glyphSizes[i] = glm::ivec2(0,0);
152  // Load first font face
153  if (FT_New_Face(ft, it->first.c_str(), 0, &face)) ENGINE_FATAL("Error: Treetype couldn't load font: {0}", it->first.c_str());
154 
155  // Set font size to load glyphs as
156  if (FT_Set_Pixel_Sizes(face, 0, it->second)) ENGINE_FATAL("Error: Freetype could not set font face size of {0}", it->second);
157 
158  for (int j = m_ASCIIstart; j <= m_ASCIIend; j++) // Drawable ASCII chars - above this we need Unicode
159  {
160  // Load char glyph and data
161  if (FT_Load_Char(face, char(j), FT_LOAD_RENDER)) ENGINE_FATAL("Error: COuld not load the character {0}", 'G');
162  int w = face->glyph->bitmap.width;
163  int h = face->glyph->bitmap.rows;
164  glyphSizes[i].x = std::max(glyphSizes[i].x, w);
165  glyphSizes[i].y = std::max(glyphSizes[i].y, h);
166  }
167 
168  int glyphNumber = 0;
169  int startX = 0;
170  int endX = glyphSizes[i].x;
171  int endY = glyphSizes[i].y;
172 
173 
174  for (int j = m_ASCIIstart; j <= m_ASCIIend; j++) // Drawable ASCII chars - above this we need Unicode
175  {
176  if (FT_Load_Char(face, char(j), FT_LOAD_RENDER)) ENGINE_FATAL("Error: COuld not load the character {0}", 'G');
177  int w = face->glyph->bitmap.width;
178  int h = face->glyph->bitmap.rows;
179  unsigned char* bmp = face->glyph->bitmap.buffer;
180  glm::ivec2 bearing(face->glyph->bitmap_left, face->glyph->bitmap_top);
181  unsigned int advance = (face->glyph->advance.x >> 6);
182 
183  // Set Character properties
184  Character currentChar(glm::ivec2(w,h),bearing,advance);
185  // Set UVs to be char size for now - divided by texture size later
186  glm::vec2 startUV = glm::vec2((float)startX, (float)startY);
187  glm::vec2 endUV = glm::vec2((float)(startX + w), (float)(startY + h));
188  currentChar.setUVs(startUV, endUV);
189  chars[j-m_ASCIIstart] = currentChar;
190 
191  // Load char glyph and data
192  for (int x = 0; x < w; x++)
193  {
194  for (int y = 0; y < h; y++)
195  {
196  texMemory[(y + startY) * memW + (x + startX)] = bmp[y * w + x];
197  }
198  }
199  // Update used area
200  usedX = std::max(usedX, startX + w);
201  usedY = std::max(usedY, startY + h);
202 
203  // increment variables
204  if (endX + glyphSizes[i].x < memW)
205  {
206  startX += glyphSizes[i].x;
207  endX += glyphSizes[i].x;
208  usedX = std::max(usedX, endX);
209  }
210  else
211  {
212  startX = 0;
213  endX = glyphSizes[i].x;
214  startY += glyphSizes[i].y;
215  endY += glyphSizes[i].y;
216 
217  if (endY > memH) ENGINE_FATAL("Run out of texture memory for fonts");
218  }
219  }
220  std::string fontName = parseFilePath(it->first);
221  m_characters[fontName] = chars;
222 
223  startY += glyphSizes[i].y;
224  endY += glyphSizes[i].y;
225 
226  if (endY > memH) ENGINE_FATAL("Run out of texture memory for fonts");
227 
228  i++;
229  }
230  unsigned char * usedMemory = (unsigned char *)malloc(usedX * usedY);
231 
232  // Copy texMemory to used Memory
233  for (int x = 0; x < usedX; x++)
234  {
235  for (int y = 0; y < usedY; y++)
236  {
237  usedMemory[(y * usedX) + x] = texMemory[(y * memW) + x];
238  }
239  }
240 
241  // Set final UV coords
242  for (auto it = m_characters.begin(); it != m_characters.end(); ++it)
243  {
244  for (auto it2 = it->second.begin(); it2 != it->second.end(); ++it2)
245  {
246  glm::vec2 startUV = it2->getstartUV();
247  glm::vec2 endUV = it2->getendUV();
248  startUV.x = startUV.x / usedX;
249  startUV.y = startUV.y / usedY;
250  endUV.x = endUV.x / usedX;
251  endUV.y = endUV.y / usedY;
252  it2->setUVs(startUV, endUV);
253  }
254  }
255 
256 
257  m_fontTexture = addTexture("FontTexture", usedX, usedY, 1, usedMemory);
258 
259  free(usedMemory);
260  free(texMemory);
261  FT_Done_Face(face);
262  FT_Done_FreeType(ft);
263  }
