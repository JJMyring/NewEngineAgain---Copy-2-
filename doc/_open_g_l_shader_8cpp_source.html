50  void OpenGLShader::parseSource(const std::string & filepath)
51  {
52  std::fstream handle(filepath, std::ios::in);
53  enum { NONE = -1, VERTEX = 0, FRAGMENT } region;
54  if (!handle.is_open()) ENGINE_FATAL("Could not open shader file '{0}'.", filepath);
55 
56  std::string line, src[2];
57 
58  std::unordered_map<std::string, ShaderDataType> uniforms; // name and types of uniforms
59 
60  while (getline(handle, line))
61  {
62  // Determine shader src type
63  if (line.find("#region Vertex") != std::string::npos) { region = VERTEX; continue; }
64  if (line.find("#region Fragment") != std::string::npos) { region = FRAGMENT; continue; }
65  if (line.find("#region Geometry") != std::string::npos) { region = NONE; continue; }
66  if (line.find("#region Tessalation") != std::string::npos) { region = NONE; continue; }
67 
68  if (region == VERTEX)
69  {
70  // Capture buffer layout
71  size_t start = line.find(" in ");
72  size_t end = line.find(";");
73  if (start != std::string::npos)
74  {
75  std::string temp, typeStr;
76  temp = line.substr(start + 4, (end - start) - 4);
77  // Trim preceding stuff
78  std::stringstream ss(temp);
79  ss >> typeStr; // GLSL type
80 
81  // Turn typeStr into an shader data type
82  ShaderDataType type = GLSLStrToSDT(typeStr);
83  m_bufferLayout.addElement(type);
84  }
85  }
86 
87  // Capture uniform variables
88  if (line.find("uniform") != std::string::npos && line.find("layout") == std::string::npos)
89  {
90  // Extract name and type
91  std::string name, typeStr;
92 
93  std::stringstream ss(line);
94  ss >> name; //"uniform"
95  ss >> typeStr; // Uniform type as a string
96  ss >> name; // Name of uniform including terminating ';'
97 
98  // Remove semicolon from name
99  name.erase(std::remove(name.begin(), name.end(), ';'), name.end());
100 
101  // Turn typeStr into an shader data type
102  ShaderDataType type = GLSLStrToSDT(typeStr);
103 
104  uniforms[name] = type;
105  }
106 
107  if (region != NONE) src[region] += (line + "\n");
108  }
109 
110  handle.close();
111 
112  compileAndLink(src[VERTEX], src[FRAGMENT]);
113 
114  bind();
115 
116  // Cache Uniform data
117  for (auto it : uniforms)
118  {
119  GLuint location = glGetUniformLocation(m_OpenGL_ID, (it.first).c_str());
120  m_uniformCache[it.first] = std::pair<ShaderDataType, GLuint>(it.second, location);
121  m_uniformLayout[it.first] = nullptr;
122  }
