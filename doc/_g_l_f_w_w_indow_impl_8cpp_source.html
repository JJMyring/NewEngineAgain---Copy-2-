149  void GLFWWindowImpl::onResize(unsigned int width, unsigned int height)
150  {
151  m_properties.m_width = width;
152  m_properties.m_height = width;
153 
154  float newAspectRatio = (float)width / (float)height;
155 
156  if (newAspectRatio < m_aspectRatio)
157  {
158  // Letterbox mode
159  float vpHeight; // Height of the viewport
160  float gap; // Gap at the top and bottom of the screen
161 
162  vpHeight = (float)width / m_aspectRatio;
163  gap = ((float)height - vpHeight) / 2.0f;
164 
165  // Get the current clear color
166  GLfloat currentClearColour[4];
167  glGetFloatv(GL_COLOR_CLEAR_VALUE, currentClearColour);
168  // Clear to a black background
169  glClearColor(0.f, 0.f, 0.f, 0.f);
170  glClear(GL_COLOR_BUFFER_BIT);
171  // Reset clear colour
172  glClearColor(currentClearColour[0], currentClearColour[1], currentClearColour[2], currentClearColour[3]);
173  // Set viewport and scissor (to limit glClear)
174  glViewport(0, (int)gap, width, (int)vpHeight);
175  glScissor(0, (int)gap, width, (int)vpHeight);
176  glEnable(GL_SCISSOR_TEST);
177  }
178  else if (newAspectRatio > m_aspectRatio)
179  {
180  // Pillarbox mode
181  float vpWidth; // Width of the viewport
182  float gap; // Gap at the top and bottom of the screen
183 
184  vpWidth = (float)height * m_aspectRatio;
185  gap = ((float)width - vpWidth) / 2.0f;
186 
187  // Get the current clear color
188  GLfloat currentClearColour[4];
189  glGetFloatv(GL_COLOR_CLEAR_VALUE, currentClearColour);
190  // Clear to a black background
191  glClearColor(0.f, 0.f, 0.f, 0.f);
192  glClear(GL_COLOR_BUFFER_BIT);
193  // Reset clear color
194  glClearColor(currentClearColour[0], currentClearColour[1], currentClearColour[2], currentClearColour[3]);
195  // Set viewport and scissor (to limit glClear)
196  glViewport((int)gap, 0, vpWidth, (int)height);
197  glScissor((int)gap, 0, vpWidth, (int)height);
198  glEnable(GL_SCISSOR_TEST);
199  }
200  else
201  {
202  glViewport(0, 0, (int)width, (int)height);
203  glScissor(0, 0, (int)width, (int)height);
